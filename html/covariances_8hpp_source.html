15template<int N> [[nodiscard]] bool is_covariance_valid(const Eigen::Matrix<double, N, N>& covariance) noexcept
16{
17 // covariance should be symetrical
18 if (!covariance.isApprox(covariance.transpose()))
19 {
20 outputs::log_warning("Covariance is not symetrical");
21 return false;
22 }
23
24 // check that this covariance is positive semi definite
25 Eigen::SelfAdjointEigenSolver<const Eigen::Matrix<double, N, N>> solver(covariance);
26 // any value < 0 is an error
27 const bool isPositiveSemiDefinite = not std::ranges::any_of(solver.eigenvalues(), [](double value) {
28 return value < 0;
29 });
30
31 if (not isPositiveSemiDefinite)
32 {
33 outputs::log_warning("Covariance is not positive semi definite");
34 }
35 return isPositiveSemiDefinite;
36}
