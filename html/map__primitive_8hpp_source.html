56 double track(const utils::PlaneWorldCoordinates& newDetectionParameters,
57 const matrix44& newDetectionCovariance,
58 const utils::WorldCoordinate& detectedCentroid)
59 {
60 assert(_kalmanFilter != nullptr);
61
62 const std::pair<vector4, matrix44>& res = _kalmanFilter->get_new_state(
63 _parametrization, _covariance, newDetectionParameters, newDetectionCovariance);
64 const vector4& newEstimatedParameters = res.first;
65 const matrix44& newEstimatedCovariance = res.second;
66
67 const double score = (_parametrization - res.first).norm();
68
69 // source: Revisiting Uncertainty Analysis for Optimum Planes Extracted from 3D Range Sensor Point-Clouds
70
71 // covariance update
72 Eigen::SelfAdjointEigenSolver<matrix44> covarianceSolver(newEstimatedCovariance);
73 const double smallestEigenValue = covarianceSolver.eigenvalues()(0);
74 const vector4& smallestEigenVector = covarianceSolver.eigenvectors().col(0).normalized();
75 _covariance =
76 newEstimatedCovariance - smallestEigenValue * smallestEigenVector * smallestEigenVector.transpose();
77
78 // parameters update
79 vector4 renormalizedVector = newEstimatedParameters / sqrt(pow(newEstimatedParameters.head(3).norm(), 2.0) +
80 pow(newEstimatedParameters(3), 2.0));
81 renormalizedVector /= sqrt(1.0 - pow(renormalizedVector(3), 2.0));
82 _parametrization << renormalizedVector.head(3).normalized(), renormalizedVector(3);
83
84 // update centroid
85 _centroid = detectedCentroid;
86
87 // static sanity checks
88 assert(_covariance.diagonal()(0) >= 0 and _covariance.diagonal()(1) >= 0 and _covariance.diagonal()(2) >= 0 and
89 _covariance.diagonal()(3) >= 0);
90 assert(not std::isnan(_parametrization.x()) and not std::isnan(_parametrization.y()) and
91 not std::isnan(_parametrization.z()) and not std::isnan(_parametrization.w()));
92 return score;
93 }

97 matrix44 _covariance; // covariance of this plane in world space
98 utils::WorldCoordinate _centroid; // centroid of the detected plane

105 static void build_kalman_filter()
106 {
107 if (_kalmanFilter == nullptr)
108 {
109 const double parametersProcessNoise = 0; // TODO set in parameters
110 const size_t stateDimension = 4; // nx, ny, nz, d
111 const size_t measurementDimension = 4; // nx, ny, nz, d
112
113 matrixd systemDynamics(stateDimension, stateDimension); // System dynamics matrix
114 matrixd outputMatrix(measurementDimension, stateDimension); // Output matrix
115 matrixd processNoiseCovariance(stateDimension, stateDimension); // Process noise covariance
116
117 // Points are not supposed to move, so no dynamics
118 systemDynamics.setIdentity();
119 // we need all positions
120 outputMatrix.setIdentity();
121
122 processNoiseCovariance.setIdentity();
123 processNoiseCovariance *= parametersProcessNoise;
124
125 _kalmanFilter = new tracking::SharedKalmanFilter(systemDynamics, outputMatrix, processNoiseCovariance);
126 }
127 }
128 // shared kalman filter, between all planes
129 inline static tracking::SharedKalmanFilter* _kalmanFilter = nullptr;

168 const int detectedPlaneSize = static_cast<int>(detectedFeatures.size());
169 for (int planeIndex = 0; planeIndex < detectedPlaneSize; ++planeIndex)
170 {
171 if (isDetectedFeatureMatched[planeIndex])
172 // Does not allow multiple removal of a single match
173 // TODO: change this
174 continue;
175
176 assert(planeIndex >= 0 and planeIndex < detectedPlaneSize);
177 const features::primitives::Plane& shapePlane = detectedFeatures[planeIndex];
178
179 // compute a similarity score
180 const double descriptorSimilarity = shapePlane.get_similarity(descriptor);
181 if (descriptorSimilarity < smallestSimilarity)
182 {
183 selectedIndex = static_cast<int>(planeIndex);
184 smallestSimilarity = descriptorSimilarity;

186 }
