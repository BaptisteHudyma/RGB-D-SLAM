231 void get_matches(const DetectedFeaturesObject& detectedFeatures,
232 const WorldToCameraMatrix& worldToCamera,
233 const bool useAdvancedMatch,
234 std::list<FeatureMatchType>& matches) noexcept
235 {
236 if (not _isActivated)
237 return;
238
239 // reset match status
240 _isDetectedFeatureMatched = vectorb::Zero(detectedFeatures.size());
241 matches.clear();
242
243 // search matches in local map first
244 for (auto& [mapId, mapFeature]: _localMap)
245 {
246 assert(mapId == mapFeature._id);
247 // start by reseting this point
248 mapFeature.mark_unmatched();
249
250 if (not mapFeature.is_visible(worldToCamera))
251 continue;
252
253 const int matchIndex = mapFeature.find_match(
254 detectedFeatures, worldToCamera, _isDetectedFeatureMatched, matches, true, useAdvancedMatch);
255 if (matchIndex >= 0)
256 {
257 mapFeature.mark_matched(matchIndex);
258 _isDetectedFeatureMatched[matchIndex] = true;
259 }
260 }
261
262 // TODO Remove this limit or make it type dependent
263 // if we have enough points from local map to run the optimization, no need to add the staged points
264 // Still, we need to try and match them to insure tracking and new map points
265 constexpr uint minimumPointsForOptimization =
266 parameters::optimization::minimumPointForOptimization *
267 3; // TODO: Why 3 ? seems about right to be sure to have enough points for the optimization process...
268 const bool shouldUseStagedPoints = matches.size() < minimumPointsForOptimization;
269
270 // search matches in staged map second
271 for (auto& [mapId, mapFeature]: _stagedMap)
272 {
273 assert(mapId == mapFeature._id);
274 // start by reseting this point
275 mapFeature.mark_unmatched();
276
277 if (not mapFeature.is_visible(worldToCamera))
278 continue;
279
280 const int matchIndex = mapFeature.find_match(detectedFeatures,
281 worldToCamera,
282 _isDetectedFeatureMatched,
283 matches,
284 shouldUseStagedPoints,
285 useAdvancedMatch);
286 if (matchIndex >= 0)
287 {
288 mapFeature.mark_matched(matchIndex);
289 _isDetectedFeatureMatched[matchIndex] = true;
290 }
291 }

362 void add_features_to_staged_map(const matrix33& poseCovariance,
363 const CameraToWorldMatrix& cameraToWorld,
364 const DetectedFeaturesObject& detectedFeatures,
365 const bool addAllFeatures) noexcept
366 {
367 if (not _isActivated)
368 return;
369
370 // Add all unmatched points to staged point container
371 const size_t featureVectorSize = detectedFeatures.size();
372 assert(featureVectorSize == static_cast<size_t>(_isDetectedFeatureMatched.size()));
373 for (unsigned int i = 0; i < featureVectorSize; ++i)
374 {
375 // Add all features, or add only the unmatched points
376 if (addAllFeatures or not _isDetectedFeatureMatched[i])
377 {
378 const DetectedFeatureType& detectedfeature = detectedFeatures.at(i);
379 // some features cannot be added to map
380 if (detectedfeature.can_add_to_map())
381 {
382 try
383 {
384 const StagedFeatureType newStagedFeature(poseCovariance, cameraToWorld, detectedfeature);
385 assert(_stagedMap.find(newStagedFeature._id) == _stagedMap.cend());
386 // add to staged map
387 _stagedMap.emplace(newStagedFeature._id, newStagedFeature);
388 }
389 catch (const std::exception& ex)
390 {
391 outputs::log_error("Caught exception while creating the staged feature: " +
392 std::string(ex.what()));
393 }
394 }
395 }
396 }

404 [[nodiscard]] bool mark_feature_with_id_as_unmatched(const size_t featureId) noexcept
405 {
406 if (not _isActivated)
407 return false;
408
409 if (featureId == 0)
410 {
411 outputs::log_error("Cannot match a feature with invalid id");
412 return false;
413 }
414 // Check if id is in local map
415 typename localMapType::iterator featureMapIterator = _localMap.find(featureId);
416 if (featureMapIterator != _localMap.end())
417 {
418 MapFeatureType& mapFeature = featureMapIterator->second;
419 assert(mapFeature._id == featureId);
420
421 if (mapFeature.is_matched())
422 {
423 const int matchindex = mapFeature._matchIndex;
424 assert(matchindex >= 0 and matchindex < _isDetectedFeatureMatched.size());
425 _isDetectedFeatureMatched[matchindex] = false;
426 mapFeature.mark_unmatched();
427 }
428 return true;
429 }
430
431 // Check if it is in staged map

433 if (stagedMapIterator != _stagedMap.end())

435 StagedFeatureType& mapFeature = stagedMapIterator->second;
436 assert(mapFeature._id == featureId);

442 _isDetectedFeatureMatched[matchindex] = false;

496 const DetectedFeaturesObject& detectedFeatureObject,
497 std::shared_ptr<outputs::IMap_Writer> mapWriter) noexcept
498 {
499 typename localMapType::iterator featureMapIterator = _localMap.begin();
500 while (featureMapIterator != _localMap.end())
501 {
502 // Update the matched/unmatched status
503 MapFeatureType& mapFeature = featureMapIterator->second;
504 assert(featureMapIterator->first == mapFeature._id);
505
506 // if matched, update the features parameters
507 if (mapFeature.is_matched())

509 assert(mapFeature._matchIndex >= 0);

514 mapFeature.update_matched(detectedFeature, poseCovariance, cameraToWorld);
515 }
516 else

518 mapFeature.update_unmatched();
519 }
520

522 {

524 mapFeature.write_to_file(mapWriter);

536 void update_staged_map(const CameraToWorldMatrix& cameraToWorld,
537 const matrix33& poseCovariance,
538 const DetectedFeaturesObject& detectedFeatureObject) noexcept
539 {
540 // Add correct staged points to local map
541 typename stagedMapType::iterator stagedFeatureIterator = _stagedMap.begin();
542 while (stagedFeatureIterator != _stagedMap.end())
543 {
544 StagedFeatureType& stagedFeature = stagedFeatureIterator->second;
545 assert(stagedFeatureIterator->first == stagedFeature._id);
546
547 // if matched, update the features parameters
548 if (stagedFeature.is_matched())
549 {
550 assert(stagedFeature._matchIndex >= 0);
551 const size_t matchedFeatureIndex = stagedFeature._matchIndex;
552 assert(matchedFeatureIndex < detectedFeatureObject.size());
553
554 const DetectedFeatureType& detectedFeature = detectedFeatureObject.at(matchedFeatureIndex);
555 stagedFeature.update_matched(detectedFeature, poseCovariance, cameraToWorld);
556 }
557 else
558 {
559 stagedFeature.update_unmatched();
560 }
561
562 if (stagedFeature.should_add_to_local_map())
563 {
564 try
565 {
566 // Add to local map, remove from staged points, with a copy of the id affected to the local map
567 _localMap.emplace(stagedFeature._id, MapFeatureType(stagedFeature));
568 assert(_localMap.at(stagedFeature._id)._id == stagedFeature._id);
569 stagedFeatureIterator = _stagedMap.erase(stagedFeatureIterator);
570 }
571 catch (const std::exception& ex)
572 {
573 outputs::log_error("Caught exeption while creating a map feature from a staged feature: " +
574 std::string(ex.what()));
575 }
576 }
577 else if (stagedFeature.should_remove_from_staged())
578 {
579 // Remove from staged points
580 stagedFeatureIterator = _stagedMap.erase(stagedFeatureIterator);
581 }
582 else
583 {
584 // Increment
585 ++stagedFeatureIterator;
586 }
587 }
