222 void get_matches(const DetectedFeaturesObject& detectedFeatures,
223 const worldToCameraMatrix& worldToCamera,
224 const bool useAdvancedMatch,
225 std::list<FeatureMatchType>& matches)
226 {
227 if (not _isActivated)
228 return;
229
230 // reset match status
231 _isDetectedFeatureMatched = vectorb::Zero(detectedFeatures.size());
232 matches.clear();
233
234 // search matches in local map first
235 for (auto& [mapId, mapFeature]: _localMap)
236 {
237 assert(mapId == mapFeature._id);
238 // start by reseting this point
239 mapFeature.mark_unmatched();
240
241 if (not mapFeature.is_visible(worldToCamera))
242 continue;
243
244 const int matchIndex = mapFeature.find_match(
245 detectedFeatures, worldToCamera, _isDetectedFeatureMatched, matches, true, useAdvancedMatch);
246 if (matchIndex >= 0)
247 {
248 mapFeature.mark_matched(matchIndex);
249 _isDetectedFeatureMatched[matchIndex] = true;
250 }
251 }
252
253 // TODO Remove this limit or make it type dependent
254 // if we have enough points from local map to run the optimization, no need to add the staged points
255 // Still, we need to try and match them to insure tracking and new map points
256 const static uint minimumPointsForOptimization =
257 Parameters::get_minimum_point_count_for_optimization() *
258 3; // TODO: Why 3 ? seems about right to be sure to have enough points for the optimization process...
259 const bool shouldUseStagedPoints = matches.size() < minimumPointsForOptimization;
260
261 // search matches in staged map second
262 for (auto& [mapId, mapFeature]: _stagedMap)
263 {
264 assert(mapId == mapFeature._id);
265 // start by reseting this point
266 mapFeature.mark_unmatched();
267
268 if (not mapFeature.is_visible(worldToCamera))
269 continue;
270
271 const int matchIndex = mapFeature.find_match(detectedFeatures,
272 worldToCamera,
273 _isDetectedFeatureMatched,
274 matches,
275 shouldUseStagedPoints,
276 useAdvancedMatch);
277 if (matchIndex >= 0)
278 {
279 mapFeature.mark_matched(matchIndex);
280 _isDetectedFeatureMatched[matchIndex] = true;
281 }

348 void add_features_to_staged_map(const matrix33& poseCovariance,
349 const cameraToWorldMatrix& cameraToWorld,
350 const DetectedFeaturesObject& detectedFeatures,
351 const bool addAllFeatures)
352 {
353 if (not _isActivated)
354 return;
355
356 // Add all unmatched points to staged point container
357 const size_t featureVectorSize = detectedFeatures.size();
358 assert(featureVectorSize == static_cast<size_t>(_isDetectedFeatureMatched.size()));
359 for (unsigned int i = 0; i < featureVectorSize; ++i)
360 {
361 // Add all features, or add only the unmatched points
362 if (addAllFeatures or not _isDetectedFeatureMatched[i])
363 {
364 const DetectedFeatureType& detectedfeature = detectedFeatures.at(i);
365 // some features cannot be added to map
366 if (detectedfeature.can_add_to_map())
367 {
368 const StagedFeatureType newStagedFeature(poseCovariance, cameraToWorld, detectedfeature);
369 assert(_stagedMap.find(newStagedFeature._id) == _stagedMap.cend());
370 // add to staged map
371 _stagedMap.emplace(newStagedFeature._id, newStagedFeature);
372 }
373 }

382 bool mark_feature_with_id_as_unmatched(const size_t featureId)

391 }
392 // Check if id is in local map
393 typename localMapType::iterator featureMapIterator = _localMap.find(featureId);
394 if (featureMapIterator != _localMap.end())

397 assert(mapFeature._id == featureId);
398
399 if (mapFeature.is_matched())

401 const int matchindex = mapFeature._matchIndex;
402 assert(matchindex >= 0 and matchindex < _isDetectedFeatureMatched.size());
403 _isDetectedFeatureMatched[matchindex] = false;
404 mapFeature.mark_unmatched();

406 return true;
407 }
408

420 _isDetectedFeatureMatched[matchindex] = false;

525 void update_staged_map(const cameraToWorldMatrix& cameraToWorld,
526 const matrix33& poseCovariance,
527 const DetectedFeaturesObject& detectedFeatureObject)
528 {
529 // Add correct staged points to local map
530 typename stagedMapType::iterator stagedFeatureIterator = _stagedMap.begin();
531 while (stagedFeatureIterator != _stagedMap.end())
532 {
533 StagedFeatureType& stagedFeature = stagedFeatureIterator->second;
534 assert(stagedFeatureIterator->first == stagedFeature._id);
535
536 // if matched, update the features parameters
537 if (stagedFeature.is_matched())
538 {
539 assert(stagedFeature._matchIndex >= 0);
540 const size_t matchedFeatureIndex = stagedFeature._matchIndex;
541 assert(matchedFeatureIndex < detectedFeatureObject.size());
542
543 const DetectedFeatureType& detectedFeature = detectedFeatureObject.at(matchedFeatureIndex);
544 stagedFeature.update_matched(detectedFeature, poseCovariance, cameraToWorld);
545 }
546 else
547 {
548 stagedFeature.update_unmatched();
549 }
550
551 if (stagedFeature.should_add_to_local_map())
552 {
553 // Add to local map, remove from staged points, with a copy of the id affected to the local map
554 _localMap.emplace(stagedFeature._id, MapFeatureType(stagedFeature));
555 assert(_localMap.at(stagedFeature._id)._id == stagedFeature._id);
556 stagedFeatureIterator = _stagedMap.erase(stagedFeatureIterator);
557 }
558 else if (stagedFeature.should_remove_from_staged())
559 {
560 // Remove from staged points
561 stagedFeatureIterator = _stagedMap.erase(stagedFeatureIterator);
562 }
563 else
564 {
565 // Increment
566 ++stagedFeatureIterator;
567 }
568 }
