17struct Point
18{
19 // world coordinates
20 utils::WorldCoordinate _coordinates;
21 // 3D descriptor (ORB)
22 cv::Mat _descriptor;
23 // position covariance
24 matrix33 _covariance;
25
26 Point(const utils::WorldCoordinate& coordinates, const matrix33& covariance, const cv::Mat& descriptor) :
27 _coordinates(coordinates),
28 _descriptor(descriptor),
29 _covariance(covariance)
30 {
31 build_kalman_filter();
32
33 assert(not _descriptor.empty() and _descriptor.cols > 0);
34 assert(not std::isnan(_coordinates.x()) and not std::isnan(_coordinates.y()) and
35 not std::isnan(_coordinates.z()));
36 };
37
46 double track(const utils::WorldCoordinate& newDetectionCoordinates, const matrix33& newDetectionCovariance)
47 {
48 assert(_kalmanFilter != nullptr);
49
50 const std::pair<vector3, matrix33>& res = _kalmanFilter->get_new_state(
51 _coordinates, _covariance, newDetectionCoordinates, newDetectionCovariance);
52
53 const double score = (_coordinates - res.first).norm();
54
55 _coordinates = res.first;
56 _covariance = res.second;
57 assert(not std::isnan(_coordinates.x()) and not std::isnan(_coordinates.y()) and
58 not std::isnan(_coordinates.z()));
59 return score;
60 }
61
62 private:
66 static void build_kalman_filter()
67 {
68 if (_kalmanFilter == nullptr)
69 {
70 // gain 10mm of uncertainty at each iteration
71 const double pointProcessNoise = 0; // TODO set in parameters
72 const size_t stateDimension = 3; // x, y, z
73 const size_t measurementDimension = 3; // x, y, z
74
75 matrixd systemDynamics(stateDimension, stateDimension); // System dynamics matrix
76 matrixd outputMatrix(measurementDimension, stateDimension); // Output matrix
77 matrixd processNoiseCovariance(stateDimension, stateDimension); // Process noise covariance
78
79 // Points are not supposed to move, so no dynamics
80 systemDynamics.setIdentity();
81 // we need all positions
82 outputMatrix.setIdentity();
83
84 processNoiseCovariance.setIdentity();
85 processNoiseCovariance *= pointProcessNoise;
86
87 _kalmanFilter = new tracking::SharedKalmanFilter(systemDynamics, outputMatrix, processNoiseCovariance);
88 }
89 }
90 // shared kalman filter, between all points
91 inline static tracking::SharedKalmanFilter* _kalmanFilter = nullptr;
92};

121 virtual int find_match(const DetectedKeypointsObject& detectedFeatures,
122 const worldToCameraMatrix& worldToCamera,
123 const vectorb& isDetectedFeatureMatched,
124 std::list<PointMatchType>& matches,
125 const bool shouldAddToMatches = true,
126 const bool useAdvancedSearch = false) const override
127 {
128 static const double searchSpaceRadius = Parameters::get_search_matches_distance();
129 static const double advancedSearchSpaceRadius = Parameters::get_search_matches_distance() * 2;
130 const double searchRadius = useAdvancedSearch ? advancedSearchSpaceRadius : searchSpaceRadius;
131
132 // try to match with tracking
133 const int invalidfeatureIndex = features::keypoints::INVALID_MATCH_INDEX;
134 int matchIndex = detectedFeatures.get_tracking_match_index(_id, isDetectedFeatureMatched);
135 if (matchIndex == invalidfeatureIndex)
136 {
137 // No match: try to find match in a window around the point
138 utils::ScreenCoordinate2D projectedMapPoint;
139 const bool isScreenCoordinatesValid = _coordinates.to_screen_coordinates(worldToCamera, projectedMapPoint);
140 if (isScreenCoordinatesValid)
141 {
142 matchIndex = detectedFeatures.get_match_index(
143 projectedMapPoint, _descriptor, isDetectedFeatureMatched, searchRadius);
144 }
145 }
146
147 if (matchIndex == invalidfeatureIndex)
148 {
149 // unmatched point
150 return UNMATCHED_FEATURE_INDEX;
151 }
152
153 assert(matchIndex >= 0);
154 assert(static_cast<Eigen::Index>(matchIndex) < isDetectedFeatureMatched.size());
155 if (isDetectedFeatureMatched[matchIndex])
156 {
157 // point was already matched
158 outputs::log_error("The requested point unique index is already matched");
159 }
160
161 if (shouldAddToMatches)
162 {
163 const rgbd_slam::screenCoordinateCovariance& screenCovariance =
164 utils::get_screen_point_covariance(_coordinates, _covariance);
165 // consider only the diagonal part of the matrix: it is the 2D variance en x/y in screen space
166 const vector2& screenPointCovariance(screenCovariance.diagonal().head(2));
167 matches.emplace_back(PointMatchType(detectedFeatures.get_keypoint(matchIndex),
168 _coordinates,
169 _covariance.diagonal(),
170 screenPointCovariance,
171 _id));
172 }
173 return matchIndex;
174 }

236 virtual bool update_with_match(const DetectedPointType& matchedFeature,
237 const matrix33& poseCovariance,
238 const cameraToWorldMatrix& cameraToWorld) override
239 {
240 assert(_matchIndex >= 0);
241
242 const utils::ScreenCoordinate& matchedScreenPoint = matchedFeature._coordinates;
243 if (utils::is_depth_valid(matchedScreenPoint.z()))
244 {
245 // transform screen point to world point
246 const utils::WorldCoordinate& worldPointCoordinates =
247 matchedScreenPoint.to_world_coordinates(cameraToWorld);
248 // get a measure of the estimated variance of the new world point
249 const cameraCoordinateCovariance& cameraPointCovariance =
250 utils::get_camera_point_covariance(matchedScreenPoint);
251 const matrix33& worldCovariance = poseCovariance + cameraPointCovariance.base();
252 // update this map point errors & position
253 track(worldPointCoordinates, worldCovariance);
254
255 // If a new descriptor is available, update it
256 const cv::Mat& descriptor = matchedFeature._descriptor;
257 if (not descriptor.empty())
258 _descriptor = descriptor;
259
260 return true;
261 }
262 else
263 {
264 // TODO: Point is 2D, handle separatly
265 }
